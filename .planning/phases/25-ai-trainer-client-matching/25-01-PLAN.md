  if (!user) return;
  const { error } = await (supabase as any)
    .from('client_profiles')
    .upsert({ user_id: user.id, ...updates }, { onConflict: 'user_id' });
  if (!error) toast.success('Saved', { duration: 1200, id: 'profile-save' });
  else toast.error('Save failed');
};
