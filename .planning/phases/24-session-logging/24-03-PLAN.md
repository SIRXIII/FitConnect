**getISOWeek helper** (private, inside same file):
```typescript
function getISOWeek(date: Date): number {
  const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
  const dayNum = d.getUTCDay() || 7;
  d.setUTCDate(d.getUTCDate() + 4 - dayNum);
  const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
  return Math.ceil((((d.getTime() - yearStart.getTime()) / 86400000) + 1) / 7);
}
```

**getISOWeekMonday helper** (private, for display label):
```typescript
function getISOWeekMonday(date: Date): Date {
  const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
  const day = d.getUTCDay() || 7;
  d.setUTCDate(d.getUTCDate() - day + 1);
  return d;
}
```

const { data, error } = await (supabase as any)
  .from('session_logs')
  .select(`
    id, booking_id, notes, exercises, created_at,
    bookings!session_logs_booking_id_fkey (
      availability_slots!bookings_slot_id_fkey (start_time, end_time),
      trainer_profiles!bookings_trainer_id_fkey (
        profiles!trainer_profiles_user_id_fkey (full_name)
